import unittest

matrix_divided = __import__('2-matrix_divided').matrix_divided
""" Import the matrix divided function from the main file """

class TestMatrixDividedFunction(unittest.TestCase):
	def test_matrix_divided(self):
		matrix = [
			[1, 2, 3],
			[4, 5, 6]
		]
		self.assertEqual(matrix_divided(matrix, 3), [[0.33, 0.67, 1.0], [1.33, 1.67, 2.0]])

	def test_matrix_divided_with_zero_divisor(self):
		matrix = [
			[1, 2, 3],
			[4, 5, 6]
		]
		with self.assertRaises(ZeroDivisionError) as context:
		matrix_divided(matrix, 0)
	self.assertEqual(str(context.exception), "division by zero")

	def test_matrix_divided_with_non_number_divisor(self):
	matrix = [
		[1, 2, 3],
		[4, 5, 6]
	]
	with self.assertRaises(TypeError) as context:
		matrix_divided(matrix, "abc")
	self.assertEqual(str(context.exception), "div must be a number")

if __name__ == '__main__':
	unittest.main()
